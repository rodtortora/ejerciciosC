Ej 7 – Dado un digrafo almacenado en una Matriz de Adyacencia, obtener de forma recursiva el
grado de entrada de cada vértice y almacenarlo en un arreglo.

void grE(int Mat[][], int i, int j, int Vec[], int N) {
	if (j<N) {
		if (i<N) {
			if (Mat[i][j] == 1) {
				Vec[j]++;			
			}
			grE(Mat[][], i+1, j, Vec[], N);
		} else {
			grE(Mat[][], 0, j+1, Vec[], N);
		}

	}
	
}
